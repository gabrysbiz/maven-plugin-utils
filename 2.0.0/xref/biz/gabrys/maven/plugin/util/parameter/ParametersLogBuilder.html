27  /**
28   * <p>
29   * Builder which allows to log Mojos parameters values. It supports basic types like arrays and collections. All other
30   * types are treat as {@link Object}s. The builder also provides API to use custom:
31   * </p>
32   * <ul>
33   * <li>{@link ValueToStringConverter} - responsible for converting parameter value to string representation</li>
34   * <li>{@link ValueSanitizer} - responsible for sanitizing Mojo parameter value</li>
35   * </ul>
36   * <p>
37   * Example:
38   * </p>
39   * 
40   * <pre>
41   * public class ExampleMojo extends AbstractMojo {
42   *
43   *     public void execute() {
44   *         logParameters();
45   *         ...
46   *     }
47   *
48   *     private void logParameters() {
49   *         ParametersLogBuilder logger = new ParametersLogBuilder(getLog());
50   *
51   *         // overwrite parameter
52   *         logger.append("parameter", 1);
53   *         logger.append("parameter", 2);
54   *
55   *         // default converter and sanitizer
56   *         logger.append("booleanParameter", true);
57   *         logger.append("stringArray", {"one", "two", "three"});
58   *
59   *         // custom converter
60   *         logger.append("customObject", new Object(), new {@link ValueToStringConverter}() {
61   *
62   *             public String convert(final Object value) {
63   *                 return "custom object";
64   *             }
65   *         });
66   *
67   *         // simple sanitizer
68   *         logger.append("booleanTrueValue", true, new {@link biz.gabrys.maven.plugin.util.parameter.sanitizer.SimpleSanitizer SimpleSanitizer}(true, Boolean.TRUE));
69   *         logger.append("booleanFalseValue", false, new {@link biz.gabrys.maven.plugin.util.parameter.sanitizer.SimpleSanitizer SimpleSanitizer}(false, Boolean.TRUE));
70   *
71   *         // lazy sanitizer
72   *         boolean condition1 = true;
73   *         logger.append("stringLazyNotExecuted", "lazy-not-executed", new {@link biz.gabrys.maven.plugin.util.parameter.sanitizer.LazySimpleSanitizer LazySimpleSanitizer}(condition1, new ValueContainer() {
74   *
75   *             public Object getValue() {
76   *                 // never executed, because condition1 is equal to true 
77   *                 return "heavyOperation1()";
78   *             }
79   *         }));
80   *         boolean condition2 = false;
81   *         logger.append("stringLazyExecuted", "lazy-executed", new {@link biz.gabrys.maven.plugin.util.parameter.sanitizer.LazySimpleSanitizer LazySimpleSanitizer}(condition2, new ValueContainer() {
82   *
83   *             public Object getValue() {
84   *                 // executed, because condition2 is equal to false 
85   *                 return "heavyOperation2()";
86   *             }
87   *         }));
88   *
89   *         // custom sanitizer
90   *         final String customSanitizerValue = null;
91   *         logger.append("customSanitizerValue", customSanitizerValue, new {@link ValueSanitizer}() {
92   *
93   *             private Object sanitizedValue;
94   *
95   *             public boolean isValid(final Object value) {
96   *                 if (customSanitizerValue != null) {
97   *                     return true;
98   *                 }
99   *                 sanitizedValue = "heavyOperation3()";
100  *                 return sanitizedValue == null;
101  *             }
102  *
103  *             public Object sanitize(final Object value) {
104  *                 return sanitizedValue;
105  *             }
106  *         });
107  *
108  *         // log data
109  *         logger.debug();
110  *     }
111  *
112  *     ...
113  * }
114  * </pre>
115  * <p>
116  * Output:
117  * </p>
118  * 
119  * <pre>
120  * [DEBUG] Job parameters:
121  * [DEBUG]       parameter = 2
122  * [DEBUG]       booleanParameter = true
123  * [DEBUG]       stringArray = ["one", "two", "three"]
124  * [DEBUG]       customObject = custom object
125  * [DEBUG]       booleanTrueValue = true
126  * [DEBUG]       booleanFalseValue = false (calculated: true)
127  * [DEBUG]       stringLazyNotExecuted = lazy-not-executed
128  * [DEBUG]       stringLazyExecuted = lazy-executed (calculated: heavyOperation2())
129  * [DEBUG]       customSanitizerValue = null (calculated: heavyOperation3())
130  * </pre>
131  * 
132  * @since 1.3.0
133  */
134 public class ParametersLogBuilder {
135 
136     /**
137      * Stores parameters names and associated containers.
138      * @since 1.3.0
139      * @see Container
140      */
141     protected final Map<String, Container> parameters;
142     /**
143      * Logger used to log parameters.
144      * @since 1.3.0
145      */
146     protected final Log logger;
147 
148     /**
149      * Constructs a new object.
150      * @param logger the logger used to log parameters.
151      * @throws IllegalArgumentException if the logger is equal to {@code null}.
152      * @since 1.3.0
153      */
154     public ParametersLogBuilder(final Log logger) {
155         this(new LinkedHashMap<String, Container>(), logger);
156     }
157 
158     /**
159      * Constructs a new object.
160      * @param parameters the map which stores parameters names and associated containers.
161      * @param logger the logger used to log parameters.
162      * @throws IllegalArgumentException if the parameters or logger is equal to {@code null}.
163      * @since 1.3.0
164      */
165     protected ParametersLogBuilder(final Map<String, Container> parameters, final Log logger) {
166         ParameterUtils.verifyNotNull("parameters", parameters);
167         ParameterUtils.verifyNotNull("logger", logger);
168 
169         this.parameters = parameters;
170         this.logger = logger;
171     }
172 
173     /**
174      * Appends a parameter with a valid value. If parameter has been appended before, then it removes its previous value
175      * and adds a new.
176      * @param name the parameter name.
177      * @param value the parameter value.
178      * @return {@code this} builder.
179      * @since 1.3.0
180      * @see DefaultValueToStringConverter
181      * @see AlwaysValidSanitizer
182      */
183     public ParametersLogBuilder append(final String name, final Object value) {
184         return append(name, value, new DefaultValueToStringConverter(), new AlwaysValidSanitizer());
185     }
186 
187     /**
188      * Appends a parameter with a valid value. If parameter has been appended before, then it removes its previous value
189      * and adds a new.
190      * @param name the parameter name.
191      * @param value the parameter value.
192      * @param converter the converter responsible for converting parameter value to string representation.
193      * @return {@code this} builder.
194      * @since 1.3.0
195      * @see AlwaysValidSanitizer
196      */
197     public ParametersLogBuilder append(final String name, final Object value, final ValueToStringConverter converter) {
198         return append(name, value, converter, new AlwaysValidSanitizer());
199     }
200 
201     /**
202      * Appends a parameter with a valid value. If parameter has been appended before, then it removes its previous value
203      * and adds a new.
204      * @param name the parameter name.
205      * @param value the parameter value.
206      * @param sanitizer the sanitizer responsible for sanitizing parameter value.
207      * @return {@code this} builder.
208      * @since 1.3.0
209      * @see DefaultValueToStringConverter
210      */
211     public ParametersLogBuilder append(final String name, final Object value, final ValueSanitizer sanitizer) {
212         return append(name, value, new DefaultValueToStringConverter(), sanitizer);
213     }
214 
215     /**
216      * Appends a parameter with a valid value. If parameter has been appended before, then it removes its previous value
217      * and adds a new.
218      * @param name the parameter name.
219      * @param value the parameter value.
220      * @param converter the converter responsible for converting parameter value to string representation.
221      * @param sanitizer the sanitizer responsible for sanitizing parameter value.
222      * @return {@code this} builder.
223      * @since 1.3.0
224      */
225     public ParametersLogBuilder append(final String name, final Object value, final ValueToStringConverter converter,
226             final ValueSanitizer sanitizer) {
227         parameters.remove(name);
228         parameters.put(name, new Container(value, converter, sanitizer));
229         return this;
230     }
231 
232     /**
233      * Logs appended parameters using info mode.
234      * @return {@code true} whether the logger logged any data, otherwise {@code false}.
235      * @since 1.3.0
236      */
237     public boolean info() {
238         return log(new InfoInternalLoggerAdapter(logger));
239     }
240 
241     /**
242      * Logs appended parameters using debug mode.
243      * @return {@code true} whether the logger logged any data, otherwise {@code false}.
244      * @since 1.3.0
245      */
246     public boolean debug() {
247         return log(new DebugInternalLoggerAdapter(logger));
248     }
249 
250     /**
251      * Logs appended parameters using {@link InternalLogger}.
252      * @param internalLogger the logger used to log parameters.
253      * @return {@code true} whether the logger logged any data, otherwise {@code false}.
254      * @since 1.3.0
255      * @see #info()
256      * @see #debug()
257      */
258     protected boolean log(final InternalLogger internalLogger) {
259         if (!internalLogger.isEnabled()) {
260             return false;
261         }
262         final List<String> lines = createLines();
263         if (lines.isEmpty()) {
264             return false;
265         }
266         for (final String line : lines) {
267             internalLogger.log(line);
268         }
269         return true;
270     }
271 
272     /**
273      * Creates all lines which will be logged by the logger.
274      * @return the lines which will be logged by the logger.
275      * @since 1.3.0
276      * @see #log(InternalLogger)
277      */
278     protected List<String> createLines() {
279         final List<String> lines = new ArrayList<String>(parameters.size() + 2);
280         lines.add("Job parameters:");
281         lines.addAll(createParametersLines());
282         lines.add("");
283         return lines;
284     }
285 
286     /**
287      * Creates one line for every appended parameter.
288      * @return the lines with parameters.
289      * @since 1.3.0
290      * @see #createLines()
291      */
292     protected List<String> createParametersLines() {
293         final List<String> lines = new ArrayList<String>(parameters.size());
294         for (final Map.Entry<String, Container> entry : parameters.entrySet()) {
295             lines.add(createParemeterLine(entry.getKey(), entry.getValue()));
296         }
297         return lines;
298     }
299 
300     /**
301      * Creates a line for single parameter.
302      * @param name the parameter name.
303      * @param container the container which stores the parameter value and associated converter and sanitizer.
304      * @return the line with the parameter name and value.
305      * @since 1.3.0
306      * @see #createParametersLines()
307      */
308     protected String createParemeterLine(final String name, final Container container) {
309         final StringBuilder line = new StringBuilder();
310         line.append("      ");
311         line.append(name);
312         line.append(" = ");
313         line.append(createParameterValue(container));
314         return line.toString();
315     }
316 
317     /**
318      * Creates a string representation of the parameter value.
319      * @param container the container which stores the parameter value and associated converter and sanitizer.
320      * @return the string representation of the parameter value.
321      * @since 1.3.0
322      * @see #createParemeterLine(String, Container)
323      */
324     protected String createParameterValue(final Container container) {
325         final StringBuilder line = new StringBuilder();
326 
327         final ValueToStringConverter converter = container.getConverter();
328         final Object value = container.getValue();
329         final String valueText = String.valueOf(converter.convert(value));
330         line.append(valueText);
331 
332         final ValueSanitizer sanitizer = container.getSanitizer();
333         if (!sanitizer.isValid(value)) {
334             final Object calculatedValue = sanitizer.sanitize(value);
335             final String calculatedValueText = String.valueOf(converter.convert(calculatedValue));
336             if (!valueText.equals(calculatedValueText)) {
337                 line.append(" (calculated: ");
338                 line.append(calculatedValueText);
339                 line.append(')');
340             }
341         }
342         return line.toString();
343     }

345     /**
346      * Container which stores parameter value and associated converter and sanitizer.
347      * @since 1.3.0
348      */
349     protected static class Container {
350 
351         private final Object value;
352         private final ValueToStringConverter converter;
353         private final ValueSanitizer sanitizer;
354 
355         /**
356          * Constructs a new instance.
357          * @param value the parameter value.
358          * @param converter the converter responsible for converting parameter value to string representation.
359          * @param sanitizer the sanitizer responsible for sanitizing parameter value.
360          * @since 1.3.0
361          */
362         protected Container(final Object value, final ValueToStringConverter converter, final ValueSanitizer sanitizer) {
363             this.value = value;
364             this.converter = converter;
365             this.sanitizer = sanitizer;
366         }
367 
368         /**
369          * Returns a parameter value.
370          * @return the parameter value.
371          * @since 1.3.0
372          */
373         protected Object getValue() {
374             return value;
375         }
376 
377         /**
378          * Returns a converter responsible for converting parameter value to string representation.
379          * @return the converter.
380          * @since 1.3.0
381          */
382         protected ValueToStringConverter getConverter() {
383             return converter;
384         }
385 
386         /**
387          * Returns a sanitizer responsible for sanitizing parameter value.
388          * @return the sanitizer.
389          * @since 1.3.0
390          */
391         protected ValueSanitizer getSanitizer() {
392             return sanitizer;
393         }
394     }

396     /**
397      * <p>
398      * Used to adaptation instance of the {@link org.apache.maven.plugin.logging.Log} to logger used in the
399      * {@link ParametersLogBuilder#log(InternalLogger)} method.
400      * </p>
401      * @since 1.3.0
402      */
403     protected interface InternalLogger {
404 
405         /**
406          * Checks whether a logger is enabled.
407          * @return {@code true} whether the logger is enabled, otherwise {@code false}.
408          * @since 1.3.0
409          */
410         boolean isEnabled();
411 
412         /**
413          * Logs a line.
414          * @param line the line to log.
415          * @since 1.3.0
416          */
417         void log(CharSequence line);
418     }
419 
420     /**
421      * Adapts {@link org.apache.maven.plugin.logging.Log} to {@link InternalLogger} API which uses info mode.
422      * @since 1.3.0
423      * @see ParametersLogBuilder#info()
424      */
425     protected static class InfoInternalLoggerAdapter implements InternalLogger {
426 
427         private final Log logger;
428 
429         /**
430          * Constructs a new instance.
431          * @param logger the adapted logger.
432          * @since 1.3.0
433          */
434         protected InfoInternalLoggerAdapter(final Log logger) {
435             this.logger = logger;
436         }
437 
438         @Override
439         public boolean isEnabled() {
440             return logger.isInfoEnabled();
441         }
442 
443         @Override
444         public void log(final CharSequence line) {
445             logger.info(line);
446         }
447     }
448 
449     /**
450      * Adapts {@link org.apache.maven.plugin.logging.Log} to {@link InternalLogger} API which uses debug mode.
451      * @since 1.3.0
452      * @see ParametersLogBuilder#debug()
453      */
454     protected static class DebugInternalLoggerAdapter implements InternalLogger {
455 
456         private final Log logger;
457 
458         /**
459          * Constructs a new instance.
460          * @param logger the adapted logger.
461          * @since 1.3.0
462          */
463         protected DebugInternalLoggerAdapter(final Log logger) {
464             this.logger = logger;
465         }
466 
467         @Override
468         public boolean isEnabled() {
469             return logger.isDebugEnabled();
470         }
471 
472         @Override
473         public void log(final CharSequence line) {
474             logger.debug(line);
475         }
476     }

13  package biz.gabrys.maven.plugin.util.io;
14  
15  /**
16   * Represents inclusion and exclusion fileset patterns format used by the {@link FileScanner}.
17   * @since 1.0
18   */
19  public enum ScannerPatternFormat {
20  
21      /**
22       * The <a href="http://ant.apache.org/">Ant</a>
23       * <a href="http://ant.apache.org/manual/dirtasks.html#patterns">patterns</a> format.
24       * @since 1.0
25       */
26      ANT,
27  
28      /**
29       * The regular expressions format.
30       * @since 1.0
31       */
32      REGEX;
33  
34      /**
35       * Returns a scanner pattern format by name (case insensitive).
36       * @param name the scanner pattern format name.
37       * @return the scanner pattern format.
38       * @since 1.0
39       * @throws IllegalArgumentException if cannot find search pattern format related with name.
40       */
41      public static ScannerPatternFormat toPattern(final String name) {
42          for (final ScannerPatternFormat pattern : values()) {
43              if (pattern.name().equalsIgnoreCase(name)) {
44                  return pattern;
45              }
46          }
47          throw new IllegalArgumentException(String.format("Cannot find search pattern format for name \"%s\"", name));
48      }
49  }

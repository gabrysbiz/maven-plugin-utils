13  package biz.gabrys.maven.plugin.util.io;
14  
15  import java.io.File;
16  
17  import org.apache.maven.plugin.logging.Log;
18  
19  /**
20   * Provides tools to log information related with files.
21   * @since 1.2
22   */
23  public final class LoggerUtils {
24  
25      private LoggerUtils() {
26          // do nothing
27      }
28  
29      /**
30       * Logs (in debug mode) information whether {@link File#isFile() normal file} is included or excluded. Logged
31       * message contains the file {@link File#getAbsolutePath() absolute path}.
32       * @param logger the logger.
33       * @param file the included or excluded file.
34       * @param included {@code true} whether file is included, otherwise {@code false}.
35       * @since 1.2
36       */
37      public static void debugInclusion(final Log logger, final File file, final boolean included) {
38          if (logger.isDebugEnabled() && file.isFile()) {
39              logger.debug((included ? "Include " : "Exclude ") + file.getAbsolutePath());
40          }
41      }
42  }

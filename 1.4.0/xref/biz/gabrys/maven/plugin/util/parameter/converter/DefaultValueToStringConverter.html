13  package biz.gabrys.maven.plugin.util.parameter.converter;
14  
15  import java.util.Collection;
16  import java.util.Map;
17  import java.util.concurrent.ConcurrentHashMap;
18  
19  /**
20   * Default {@link ValueToStringConverter} which allows to converts basic types. Supported types:
21   * <ul>
22   * <li>arrays (see {@link ArrayToStringConverter})</li>
23   * <li>collections (see {@link CollectionToStringConverter})</li>
24   * <li>objects (see {@link ObjectToStringConverter})</li>
25   * </ul>
26   * @since 1.3.0
27   */
28  public class DefaultValueToStringConverter extends AbstractValueToStringConverter {
29  
30      private final Map<Class<?>, ValueToStringConverter> converters;
31      private final ValueToStringConverter fallback;
32  
33      /**
34       * Constructs a new instance.
35       * @since 1.3.0
36       */
37      public DefaultValueToStringConverter() {
38          converters = new ConcurrentHashMap<Class<?>, ValueToStringConverter>();
39          converters.put(Object[].class, new ArrayToStringConverter());
40          converters.put(Collection.class, new CollectionToStringConverter());
41          fallback = new ObjectToStringConverter();
42      }
43  
44      // for tests
45      DefaultValueToStringConverter(final Map<Class<?>, ValueToStringConverter> converters, final ValueToStringConverter fallback) {
46          this.converters = converters;
47          this.fallback = fallback;
48      }
49  
50      @Override
51      public String convert2(final Object value) {
52          for (final Map.Entry<Class<?>, ValueToStringConverter> entry : converters.entrySet()) {
53              if (entry.getKey().isInstance(value)) {
54                  return entry.getValue().convert(value);
55              }
56          }
57          return fallback.convert(value);
58      }
59  }

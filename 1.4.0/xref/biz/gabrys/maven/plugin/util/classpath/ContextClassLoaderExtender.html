13  package biz.gabrys.maven.plugin.util.classpath;
14  
15  import java.net.MalformedURLException;
16  import java.net.URL;
17  import java.net.URLClassLoader;
18  import java.util.ArrayList;
19  import java.util.Arrays;
20  import java.util.Collection;
21  import java.util.HashSet;
22  import java.util.LinkedList;
23  import java.util.List;
24  import java.util.Set;
25  
26  import org.apache.maven.artifact.Artifact;
27  import org.apache.maven.plugin.logging.Log;
28  import org.apache.maven.project.MavenProject;
29  
30  import biz.gabrys.maven.plugin.util.parameter.ParameterUtils;
31  
32  /**
33   * Responsible for adding Maven project dependencies to classpath.
34   * @since 1.4.0
35   */
36  public class ContextClassLoaderExtender {
37  
38      private final MavenProject project;
39      private final Log logger;
40  
41      /**
42       * Creates a new instance.
43       * @param project the Maven project.
44       * @param logger the logger.
45       * @throws IllegalArgumentException if the Maven project or/and the logger is equal to {@code null}.
46       * @since 1.4.0
47       */
48      public ContextClassLoaderExtender(final MavenProject project, final Log logger) {
49          ParameterUtils.verifyNotNull("project", project);
50          ParameterUtils.verifyNotNull("logger", logger);
51          this.project = project;
52          this.logger = logger;
53      }
54  
55      /**
56       * Adds all dependencies with specified types to context class loader.
57       * @param types the supported types.
58       * @throws IllegalArgumentException if the types are equal to {@code null}.
59       * @since 1.4.0
60       */
61      public void addDependencies(final String... types) {
62          ParameterUtils.verifyNotNull("types", types);
63          addDependencies(new HashSet<String>(Arrays.asList(types)));
64      }
65  
66      /**
67       * Adds all dependencies with specified types to context class loader.
68       * @param types the supported types.
69       * @throws IllegalArgumentException if the types collection is equal to {@code null}.
70       * @since 1.4.0
71       */
72      public void addDependencies(final Collection<String> types) {
73          ParameterUtils.verifyNotNull("types", types);
74          @SuppressWarnings("unchecked")
75          final Set<Artifact> artifacts = project.getArtifacts();
76          final List<Artifact> filtered = filterArtifacts(artifacts, types);
77          final List<URL> urls = resolveArtifactsUrls(filtered);
78          addToContextClassLoader(urls);
79      }
80  
81      /**
82       * Filters artifacts based on the type.
83       * @param artifacts the collection which stores artifacts.
84       * @param types the supported types.
85       * @return the list with artifacts whose types fit to the supported types.
86       * @since 1.4.0
87       */
88      protected List<Artifact> filterArtifacts(final Collection<Artifact> artifacts, final Collection<String> types) {
89          final List<Artifact> filtered = new LinkedList<Artifact>();
90          for (final Artifact artifact : artifacts) {
91              if (types.contains(artifact.getType())) {
92                  if (logger.isDebugEnabled()) {
93                      logger.debug(String.format("Include %s", createDisplayText(artifact)));
94                  }
95                  filtered.add(artifact);
96              } else if (logger.isDebugEnabled()) {
97                  logger.debug(String.format("Exclude %s", createDisplayText(artifact)));
98              }
99          }
100         return filtered;
101     }
102 
103     /**
104      * Returns URLs whose represents artifacts.
105      * @param artifacts the collection which stores artifacts.
106      * @return the artifacts' URLs.
107      * @since 1.4.0
108      */
109     protected List<URL> resolveArtifactsUrls(final Collection<Artifact> artifacts) {
110         final List<URL> urls = new ArrayList<URL>(artifacts.size());
111         for (final Artifact artifact : artifacts) {
112             try {
113                 urls.add(artifact.getFile().toURI().toURL());
114             } catch (final MalformedURLException e) {
115                 // never
116                 throw new IllegalStateException(String.format("Cannot add %s to the classpath!", createDisplayText(artifact)), e);
117             }
118         }
119         return urls;
120     }
121 
122     /**
123      * Adds artifacts URLs to context class loader.
124      * @param urls the artifacts URLs.
125      * @since 1.4.0
126      */
127     protected void addToContextClassLoader(final List<URL> urls) {
128         final Thread currentThread = Thread.currentThread();
129         final URLClassLoader classLoader = new URLClassLoader(urls.toArray(new URL[0]), currentThread.getContextClassLoader());
130         currentThread.setContextClassLoader(classLoader);
131     }
132 
133     /**
134      * Creates a text representation of the {@link Artifact}.
135      * @param artifact the artifact.
136      * @return the text representation of the {@link Artifact}.
137      * @since 1.4.0
138      */
139     protected String createDisplayText(final Artifact artifact) {
140         return String.format("%s:%s-%s.%s (%s)", artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion(), artifact.getType(),
141                 artifact.getScope());
142     }
143 }
